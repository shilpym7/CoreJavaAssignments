package myFirstPackage;

public class LoOp {
     public static void main(String args[])
     {
    	 int a=10,b=20,c=5;
    	 System.out.println((a>b)&&(a>c));
    	 System.out.println((a>b)||(a>c));
    	 System.out.println(!((a>b)||(a>c)));
     }
}
